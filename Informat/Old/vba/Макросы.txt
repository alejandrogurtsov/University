Sub DeleteFirstSentenceInParagraphs()
    Dim par As Paragraph
    Dim firstSent As Range
    
    For Each para In ActiveDocument.Paragraphs

        If Len(para.Range.Text) > 1 Then
            Set firstSent = para.Range.Sentences(1)
            firstSent.Delete
        End If
    Next para
End Sub

Sub CenterAllPictures()
    Dim ils As InlineShape

    For Each ils In ActiveDocument.InlineShapes
        ils.Range.ParagraphFormat.Alignment = wdAlignParagraphCenter
    Next ils
End Sub

Sub ApplyStylesToEvenOddParagraphs()
    Dim para As Paragraph
    Dim i As Long
    
    i = 1
    For Each para In ActiveDocument.Paragraphs
    
        If Len(Trim(para.Range.Text)) > 1 Then
            If i Mod 2 = 0 Then
                para.Range.Style = ActiveDocument.Styles("Строгий")
            Else
                para.Range.Style = ActiveDocument.Styles("Обычный")
            End If
            i = i + 1
        End If
    Next para
End Sub
